import { NgModule, ModuleWithProviders } from '@angular/core';

  CalendarModule,
  SliderModule,
  ChipsModule,
  PickListModule

    CommonModule,
    FormsModule,
    ReactiveFormsModule,
    InputTextModule,
    InputTextareaModule,
    ButtonModule,

    RestAdminComponent

  static forRoot(): ModuleWithProviders {
    return {
      ngModule: RestAdminModule,
    };
  }
}
